            <a [routerLink]="[ '/goodsinfo', '666' ]">标签跳转路由 到 goodsinfo产品详情页</a>
标签跳转路由 到 goodsinfo产品详情页

2、js 普通路由、动态路由跳转：

        
            // 1、引入Router模块
            import  Router } from "@angular/router"

            // 2、声明Router对象
            constructor(private router: Router)

            // 3.1、js普通路由跳转
            this.router.navigate(['/goodsinfo']);
            
            // 3.2、js动态 传参 路由跳转
            this.router.navigate(['/goodsinfo', '123']);


js get跳转路由跳转路由：

        
            // 1、引入Router模块
            import  Router, NavigationExtras } from "@angular/router"

            // 2、声明Router对象
            constructor(private router: Router)

            // 3、js get 传参 路由跳转
            let parObj: NavigationExtras = 
                queryParams:  gid: 456, rmb: '￥998.00', num: 999 }
            }
            this.router.navigate(['/goodsinfo'], parObj);    // this.router.navigate(['/goodsinfo'], NavigationExtras);
