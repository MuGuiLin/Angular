directive 自定义标签，属性，类名

			
			
			html:
			《mupiao》 《/mupiao》
			《div mupiao》 《/div》
			《div class="mupiao"》 《/div》
			《!-- directive:mupiao --》//注：中间有空格的哦！
			《div》 《/div》
			
			var APP = angular.module('MyModule' ,[]);
			APP.directive('mupiao' , function(){
				return {
					restrict : "AEMC", //A：属性，E：元素，M：注释，C：样式类
					template : "《div》我是自定标签哦，实际上我是DIV!《/div》",
					replace : true
				}
			});
