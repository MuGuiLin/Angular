$interval(简隔定时)，$timeout 定时器 (内置服务对象)

			在AngularJS中$interval用来处理间歇性处理一些事情。
			
			var APP = angular.module('APP',[])      //定时执行    超时执行
			APP.controller('test',function($scope , $interval , $timeout) 
			{
				$scope.time = 0; 
				$scope.mupiao = "";
				$interval(function(){
					$scope.time ++;
				},1000); //隔1钟执行一次
				
				$timeout(function(){
					$scope.mupiao = "我隔了5秒钟才弹出来，我只执行一次";
					alert("我隔了5秒钟才弹出来，我只执行一次");
				},5000);
			});
$interval简隔定时：{{time}}【每简隔多少时间执行一次】
$timeout超时执行：{{mupiao}}【多少时间后只执行一次】
